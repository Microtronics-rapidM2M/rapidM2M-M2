//===
//=== Measurement Channels
//===
<split>
  source =rm2mraw     //Raw data channel (always "rm2mraw" for the rapidM2M M3) from which the data records should be copied 
  target =histdata0   //Measurement data channel to which the data record should be copied
  key    =00          //If the first byte of the data record corresponds to this value, it is copied to the specified measurement data channel.
</split>
 
<table>
  name   =histdata0   //Measurement data channel 0 should be used.
  <field>
    name   =ch0       //Data field 0 should be used for the input voltage Vin.
    title  =Vin       //Title of the data field displayed on the server
    units  =mV        //Measurement value unit displayed on the server
    type   =u32       //Data type: 32Bit unsigned
    byteofs=1         //Position in measurement data channel 0 where the data field is saved
	              //Note: The key for the split tag is located at position "0" 
  </field>
  <field>
    name   =ch1       //Data field 1 should be used for the Auxiliary voltage Vaux.
    title  =Vaux      //Title of the data field displayed on the server
    units  =mV        //Measurement value unit displayed on the server
    type   =u32       //Data type: 32Bit unsigned
    byteofs=5         //Position in measurement data channel 0 where the data field is saved
  </field>
</table>


//===
//=== Settings
//===
<table>
  name   =rm2mconfig0               //Configuration block 0 should be used
  <field>
    name   =field0                  //Parameter 0 should be used for the record interval  
    title  =Record Interval         //Title of the parameter displayed on the server
    units  =Min                     //Unit to be used to enter the value for the parameter
    decpl  =0                       //No decimal places 
    vscale =0.01666666667           //Conversion factor sec.->min. (sec. used internally) 
    type   =u32                     //Data type: 32Bit unsigned
    byteofs=0                       //Position in config block 0 where the parameter is saved
  </field>
  <field>
    name   =field1                  //Parameter 1 should be used for the transmission interval
    title  =Transmission Interval   //Title of the parameter displayed on the server
    units  =min                     //Unit to be used to enter the value for the parameter
    decpl  =0                       //No decimal places 
    vscale =0.01666666667           //Conversion factor sec.->min. (sec. used internally) 
    type   =u32                     //Data type: 32Bit unsigned
    byteofs=4                       //Position in config block 0 where the parameter is saved 
  </field>
</table>
